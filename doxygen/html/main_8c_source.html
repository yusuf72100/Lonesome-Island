43Vecteur InitVecteur(int angle, int vitesse)
44{
45 Vecteur Vecteur;
46 double bufferX;
47 double bufferY;
48
49 bufferX = cos(angle) * vitesse;
50 bufferY = sin(angle) * vitesse;
51
52 Vecteur.x = round(bufferX);
53 Vecteur.y = round(bufferY);
54
55 if(debug) printf("Angle : %d\n", angle);
56 if(debug) printf("Vecteur.x : %lf pour un cos de %lf\n", Vecteur.x, cos(angle));
57 if(debug) printf("Vecteur.y : %lf pour un sin de %lf\n", Vecteur.y, sin(angle));
58
59 return Vecteur;
60}
